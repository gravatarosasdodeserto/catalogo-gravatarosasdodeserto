async function cadastrar(){ const nome = document.getElementById('nome').value.trim().toUpperCase(); const qtd = parseInt(document.getElementById('qtd').value) || 0; const senha = document.getElementById('senha').value; if(!nome) return alert('Preenche o nome'); document.getElementById('btn-cadastrar').disabled = true; mostrarStatus('Enviando... aguarda 30s', 'carregando'); try { const res = await fetch('/.netlify/functions/cadastrar', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ nome, qtd, senha }) }); if(res.ok){ mostrarStatus('Sucesso! Site vai atualizar em ~30s', 'sucesso'); document.getElementById('nome').value = ''; document.getElementById('qtd').value = '0'; document.getElementById('senha').value = ''; } else { const erro = await res.text(); throw new Error(erro); } } catch(err) { mostrarStatus('Erro: ' + err.message, 'erro'); } finally { document.getElementById('btn-cadastrar').disabled = false; } }
